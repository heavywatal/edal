163 void Simulation::life_cycle() {
164  std::vector<std::vector<Patch> > parents(population);
165  wtl::Semaphore sem(PPN);
166  std::mutex mtx;
167  auto patch_task = [&](const size_t row, const size_t col) {
168  auto offsprings = parents[row][col].mate_and_reproduce();
169  std::lock_guard<std::mutex> lck(mtx);
170  for (auto& child: offsprings) {
171  size_t dest_row = row;
172  size_t dest_col = col;
173  choose_patch(&dest_row, &dest_col);
174  population[dest_row][dest_col].append(std::move(child));
175  }
176  sem.unlock();
177  };
178  std::vector<std::thread> threads;
179  for (size_t row=0; row<NUM_ROWS; ++row) {
180  for (size_t col=0; col<NUM_COLS; ++col) {
181  sem.lock();
182  threads.emplace_back(patch_task, row, col);
183  }
184  }
185  for (auto& th: threads) {th.join();}
186  threads.clear();
187  for (size_t row=0; row<NUM_ROWS; ++row) {
188  for (size_t col=0; col<NUM_COLS; ++col) {
189  sem.lock();
190  threads.emplace_back([row, col, &sem, this] {
191  population[row][col].viability_selection();
192  sem.unlock();
193  });
194  }
195  }
196  for (auto& th: threads) {th.join();}
197 }

199 void Simulation::choose_patch(size_t* row, size_t* col) const {
200  if (!prandom().bernoulli(Individual::MIGRATION_RATE())) {return;}
201  size_t r = *row;
202  size_t c = *col;
203  switch (prandom().randrange(8)) {
204  case 0: ++c; break;
205  case 1: ++r; ++c; break;
206  case 2: ++r; break;
207  case 3: ++r; --c; break;
208  case 4: --c; break;
209  case 5: --r; --c; break;
210  case 6: --r; break;
211  case 7: --r; ++c; break;
212  }
213  // size_t(-1) also makes true
214  if ((r >= NUM_ROWS) | (c >= NUM_COLS)) {return;}
215  *row = r; *col = c;
216 }
