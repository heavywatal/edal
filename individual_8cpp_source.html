188 double Individual::calc_DI_analytical() const {
189  const double a = BETA_PARAM_;
190  const double h0 = HEIGHT_PREFERENCE_;
191  const double h1 = DIAMETER_PREFERENCE_;
192  const double y0 = phenotype_[trait::height_preference];
193  const double y1 = phenotype_[trait::diameter_preference];
194  double d = 12;
195  d -= 6*h0*(1 + 2*(-1 + y0)*y0);
196  d += h1*(-1 + 4*(1 - 3*y1)*y1);
197  d -= a*(-24 + h1 + 6*h0*wtl::pow<2>(1 - 2*y0) - 8*h1*y1 + 24*h1*wtl::pow<2>(y1));
198  d /= (12 + 24*a);
199  return d;

336 double Individual::mating_preference(const Individual& male) const {
337  const double choosiness = phenotype_[trait::choosiness];
338  if (choosiness == 0.5) {
339  // random mating
340  return 1.0;
341  }
342  double exponent = phenotype_[trait::female_trait];
343  if (choosiness > 0.5) {
344  // assortative mating
345  exponent -= male.phenotype_[trait::male_trait];
346  } else {
347  // disassortative mating
348  exponent -= 1.0;
349  exponent += male.phenotype_[trait::male_trait];
350  }
351  double true_choosiness = choosiness;
352  true_choosiness *= 2.0;
353  true_choosiness -= 1.0;
354  exponent *= true_choosiness;
355  exponent /= MATING_SIGMA_;
356  exponent *= exponent;
357  exponent *= -0.5;
358  return std::exp(exponent);
359 }

371  static const std::bitset<8> mask(MUTATION_MASK_);
372  const double mu_trait = MU_LOCUS_ * NUM_LOCI_;
373  std::vector<Loci> gamete;
