183 inline std::pair<size_t, size_t> choose_patch(size_t row, size_t col) {
184  if (!std::bernoulli_distribution(Individual::MIGRATION_RATE())(wtl::sfmt()))
185  {return {row, col};}
186  switch (std::uniform_int_distribution<size_t>(0, 7)(wtl::sfmt())) {
187  case 0: ++col; break;
188  case 1: ++row; ++col; break;
189  case 2: ++row; break;
190  case 3: ++row; --col; break;
191  case 4: --col; break;
192  case 5: --row; --col; break;
193  case 6: --row; break;
194  case 7: --row; ++col; break;
195  }
196  return {row, col};
197 }

199 void Simulation::life_cycle() {
200  std::vector<std::vector<Patch> > parents(population);
201  auto reproduction = [&](const size_t row, const size_t col) {
202  auto offsprings = parents[row][col].mate_and_reproduce(wtl::sfmt());
203  std::vector<std::pair<size_t, size_t> > destinations;
204  destinations.reserve(offsprings.size());
205  for (size_t i=0; i<offsprings.size(); ++i) {
206  const auto dst = choose_patch(row, col);
207  if ((dst.first >= NUM_ROWS) | (dst.second >= NUM_COLS)) {
208  destinations.emplace_back(row, col);
209  } else {
210  destinations.push_back(std::move(dst));
211  }
212  }
213  for (size_t i=0; i<offsprings.size(); ++i) {
214  const auto& dst = destinations[i];
215  population[dst.first][dst.second].append(std::move(offsprings[i]));
216  }
217  };
218  for (size_t row=0; row<NUM_ROWS; ++row) {
219  for (size_t col=0; col<NUM_COLS; ++col) {
220  reproduction(row, col);
221  }
222  }
223  for (size_t row=0; row<NUM_ROWS; ++row) {
224  for (size_t col=0; col<NUM_COLS; ++col) {
225  population[row][col].viability_selection(wtl::sfmt());
226  }
227  }
228 }
