171 void Simulation::life_cycle() {
172  std::vector<std::vector<Patch> > parents(population);
173  auto reproduction = [&](const size_t row, const size_t col) {
174  auto offsprings = parents[row][col].mate_and_reproduce();
175  auto& patch = population[row][col];
176  std::vector<std::pair<size_t, size_t> > destinations;
177  destinations.reserve(offsprings.size());
178  for (size_t i=0; i<offsprings.size(); ++i) {
179  const auto dst = patch.choose_patch(row, col);
180  if ((dst.first >= NUM_ROWS) | (dst.second >= NUM_COLS)) {
181  destinations.emplace_back(row, col);
182  } else {
183  destinations.push_back(std::move(dst));
184  }
185  }
186  for (size_t i=0; i<offsprings.size(); ++i) {
187  const auto& dst = destinations[i];
188  population[dst.first][dst.second].append(std::move(offsprings[i]));
189  }
190  };
191  for (size_t row=0; row<NUM_ROWS; ++row) {
192  for (size_t col=0; col<NUM_COLS; ++col) {
193  reproduction(row, col);
194  }
195  }
196  for (size_t row=0; row<NUM_ROWS; ++row) {
197  for (size_t col=0; col<NUM_COLS; ++col) {
198  population[row][col].viability_selection();
199  }
200  }
201 }
